void getNext(char s[], int len, char next[])
{
    int j = -1;
    next[0] = -1;
    for (int i = 1; i < len; ++i)
    {
        while (j != -1 && s[i] != s[j + 1])
        {
            j = next[j];
        }
        if (s[i] == s[j + 1])
        {
            j++;
        }
        next[i] = j;
    }
}
